CREATE TABLE genre_list (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    title TEXT NOT NULL,
    movie_id INTEGER,
    FOREIGN KEY (movie_id) REFERENCES Ratings (id)
);

CREATE TABLE Ratings (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    director TEXT,
    popularity REAL,
    imdb_score REAL
);
